link Arc::Std;

namespace Arc::Program {
	public enum HttpMethods {
		GET,
		POST,
		PUT,
		PATCH,
		DELETE
	}

	@Export
	public group ArcExample {
		@Getter
		public field const foo: string;
		@Accessor
		public field var bar: string;

		private func eval(): bool {
			return false;
		}

		public func empty(): none {}

		public func ctor(): none {
			foo = "foo";
			bar = "bar";
		}
	}

	@Export
	@Entrypoint
	public func main(var args: string[]): none {
		var a: int;
		a = [Console].readLn();
		var b: infer;
		a = 4;
		b = 6;
		const c: infer;
		if (a > b) { c = a + b; }
		elif (a < b) { c = a * b; }
		else { c = a - b; }

		call [Console].PrintLn("Hello, world!");

		var d: ArcExample;
		d = new ArcExample();

		return 0;
	}
}
