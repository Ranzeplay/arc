link Arc::Std;

namespace Arc::Program {
	@Export
	public group ArcExample {
		@Getter
		public field const foo: val string;
		@Accessor
		public field var bar: val string;

		private method eval(): bool {
			return false;
		}

		public function empty(): none {}
	}

	@Export
	@Entrypoint
	public function main(var args: val string[]): none {
		var a: val int;
		a = Console::readln();
		var b: val auto;
		a = 4;
		b = 6;
		const c: val auto;
		if (a > b) { c = a + b; }
		elif (a < b) { c = a * b; }
		else { c = a - b; }

		call Console::PrintLn($"abc {c} {args[0]}");

		return 0;
	}
}
